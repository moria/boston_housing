            train_y = y[(start+test_len):]
                            
        estimator.fit(train_x, train_y)
        predict_y = estimator.predict(test_x)
        score = scoring(test_y, predict_y)               
        scores.append(score)
        i = i - 1
        
    return scores

        scores = cross_val_score(clf, X, y)
        tot = 0
        for i in scores:
            tot = tot + i
            if tot >= max_tot:
                max_tot = tot
                best_estimator = clf
    
    return best_estimator
